[docs]class Actionable(object):
    __metaclass__ = ActionableMeta

    def __init__(self):
        self._callbacks = defaultdict(dict)
        self.cid = 0

    def _actioned(self, key):
        for f, args, kw in self._callbacks[key].values():
            f(*args, **kw)

    def _on_action(self, key, f, args=None, kw=None):
        args = args if args is not None else tuple()
        kw = kw if kw is not None else dict()
        cid = self.cid
        self._callbacks[key][cid] = (f, args, kw)
        self.cid += 1
        return cid

    def _disconnect_action(self, key, cid):
        try:
            del self._callbacks[key][cid]
        except KeyError:
            pass
